Юров программирование на ассемблере

задан массив целых чисел, известно их количество
посчитать кол-во положительных, прдполагая, что их не больше 9

слово в 16битной машине - 2 байте
5 слов (массив) - 10 байт

si - source index, для пробегания по массиву (16 bit)

dx - два куска ah и al (по 8 bit)
dl - для вывода одного байта на экран (8 bit)

cx - count ...


dseg segment							;dseg - название сегмента
	arr dw 3, 2, 0, -4, 7				;dw - define words, задать словами; 5 слов (массив) - 10 байт
	mes db 'полож. чисел', 13, 10, '$'	;db - define bytes
dseg ends
cseg segment							;cseg - название сегмента
	assume cs:cseg, ds:dseg				;assume - предположить
n dw 5									;в массиве dw 5 чисел
start:		mov ax, dseg					;двоеточие на конце - метка, ax на dseg
			mov ds, ax						;ds на ax
			mov si, 0						;положить в si ноль
			mov dl, 0
			mov cx, n
			cmp arr[si], 0					;сравнить arr[si] с 0
			jle next_num					;если меньше или равно 0, ничего не делать
			inc dl							;увеличить счётчик dl
next_num: 	add si, 2						;прибавить к индексатору 2 (т.к. надо перепрыгнуть 2 байта)
			dec cx
			jnz beg							;jump (if) not zero

;на вывод
;3 - полож. чисел
			add dl, 48
			mov ah, 6						;шестая фя
			int 21h							;шестая фя 21го прерывания отдаёт один символ из dl на экран
			lea dx, mes						;в dx положить эффективный адрес mes
			mov ah, 9
			int 21h							;21я фя 9 подфункция
			mov ax, 4C00h					;4C - содержимое ah, 00 - содержимое al
			int 21h
cseg ends
	end start
